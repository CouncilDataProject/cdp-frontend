import { action } from "@storybook/addon-actions";
import { Meta, Props, Story } from "@storybook/addon-docs/blocks";
import { array, select, withKnobs } from "@storybook/addon-knobs";

import SelectSorting from "./SelectSorting";

<Meta title="Filters|SelectSorting" component={SelectSorting} />

# Select Sorting

### Display radio checkboxes to select sorting options

The width of SelectSorting takes on the width of its container.

<Props of={SelectSorting} />

#### Sort Order is disabled because Sort By doesn't have a selected value.

<Story name="Disabled Sort Order">
  <SelectSorting
    state={{
      by: "",
      order: "",
    }}
    update={action("update-sorting-state")}
    sortByOptions={["name", "date", "value"]}
  />
</Story>

#### Sort Order is enabled because Sort By has a selected value.

<Story name="Enabled Sort Order">
  <SelectSorting
    state={{
      by: "name",
      order: "",
    }}
    update={action("update-sorting-state")}
    sortByOptions={["name", "date", "value"]}
  />
</Story>

### Playground

#### Experiment with these stories with Knobs addon in Canvas mode.

Change the values of the sort state to change what sort options are selected.

<Story name="Sort State Playground" parameters={{ decorators: [withKnobs] }}>
  <SelectSorting
    state={{
      by: select("Sort By", {
        "Committee Name": "name",
        "Event Date": "date",
        "Search Relevance": "value",
      }),
      order: select("Sort Order", { Ascending: "asc", Descending: "desc" }),
    }}
    update={action("update-sorting-state")}
    sortByOptions={["name", "date", "value"]}
  />
</Story>

Change the values of sort by options array to change what sort by options are visible.

<Story name="Sort By Options Playground" parameters={{ decorators: [withKnobs] }}>
  <SelectSorting
    state={{
      by: "",
      order: "",
    }}
    update={action("update-sorting-state")}
    sortByOptions={array("Sort By Options", ["name", "date", "value"])}
  />
</Story>
